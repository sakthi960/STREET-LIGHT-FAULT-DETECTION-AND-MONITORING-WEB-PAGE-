    <script>
        let voltageChart, currentChart;
        let sensorFailureCount = {};
        let isBackendConnected = false;
        let voltageHistory = [];
        let currentHistory = [];
        let timeLabels = [];

        function generateFluctuatingVoltage() {
            return (Math.random() * (11.9 - 10.0) + 10.0).toFixed(1);
        }

        function generateFluctuatingCurrent() {
            return (Math.random() * (1.0 - 0.2) + 0.2).toFixed(2);
        }

        async function controlLight(lightId, action) {
            try {
                const response = await fetch('/control', {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({light_id: lightId, action: action})
                });

                const data = await response.json();

                if (data.success) {
                    console.log(Light ${lightId} ${action} successful);
                    fetchData();
                } else {
                    alert('Control failed: ' + (data.message || 'unknown error'));
                }
            } catch (error) {
                console.error('Control error:', error);
                alert('Failed to control light. Backend not connected.');
            }
        }

        async function fetchData() {
            try {
                const response = await fetch('/api/data');
                const data = await response.json();
                
                isBackendConnected = true;
                document.getElementById('backend-status').innerHTML = 'ðŸŸ¢ Backend: Online';
                
                updateLights(data.lights);
                
                document.getElementById('last-update-text').innerHTML = 
                    <strong>Last Updated:</strong> ${data.time} | <strong>Auto-refresh:</strong> Every 2 seconds;
                
                updateCharts();
                
            } catch (error) {
                console.error('Fetch error:', error);
                isBackendConnected = false;
                document.getElementById('backend-status').innerHTML = 'ðŸ”´ Backend: Offline';
                
                resetToZeroValues();
            }
        }

        function resetToZeroValues() {
            document.getElementById('total-voltage').innerHTML = '0<span class="stat-unit">V</span>';
            document.getElementById('total-current').innerHTML = '0<span class="stat-unit">A</span>';
            document.getElementById('total-lux').innerHTML = '0<span class="stat-unit">Lux</span>';
            document.getElementById('system-status').textContent = 'Offline';
            document.getElementById('system-status').style.color = '#ff4444';

            for (let i = 1; i <= 4; i++) {
                document.getElementById(voltage-${i}).textContent = '0V';
                document.getElementById(current-${i}).textContent = '0A';
                document.getElementById(lux-${i}).textContent = '0';
                
                const bulb = document.getElementById(bulb-${i});
                bulb.className = 'light-bulb light-off';
                bulb.textContent = 'â—‹';
                
                const statusBadge = document.getElementById(status-${i});
                statusBadge.className = 'status-badge status-off';
                statusBadge.textContent = 'OFF';
                
                document.getElementById(sensor-alert-${i}).classList.remove('show');
                document.getElementById(light-card-${i}).classList.remove('sensor-failed');
            }

            if (voltageChart) {
                voltageChart.data.datasets[0].data = [0, 0, 0, 0, 0, 0];
                voltageChart.update();
            }
            if (currentChart) {
                currentChart.data.datasets[0].data = [0, 0, 0, 0, 0, 0];
                currentChart.update();
            }
        }

        function checkSensorFailure(lightId, lux) {
            if (!sensorFailureCount[lightId]) {
                sensorFailureCount[lightId] = 0;
            }

            if (lux === -1 || lux === null || lux === undefined) {
                sensorFailureCount[lightId]++;
            } else {
                sensorFailureCount[lightId] = 0;
            }

            const isFailed = sensorFailureCount[lightId] >= 2;
            const alertElement = document.getElementById(sensor-alert-${lightId});
            const cardElement = document.getElementById(light-card-${lightId});

            if (isFailed) {
                alertElement.classList.add('show');
                cardElement.classList.add('sensor-failed');
            } else {
                alertElement.classList.remove('show');
                cardElement.classList.remove('sensor-failed');
            }

            return isFailed;
        }

        function updateLights(lights) {
            let totalLux = 0;
            let totalVoltage = 0;
            let totalCurrent = 0;

            for (let i = 1; i <= 4; i++) {
                const lightKey = light${i};
                const light = lights[lightKey];

                if (light) {
                    const isOn = light.status === 'ON';

                    const bulb = document.getElementById(bulb-${i});
                    bulb.className = isOn ? 'light-bulb light-on' : 'light-bulb light-off';
                    bulb.textContent = isOn ? 'â—' : 'â—‹';

                    let voltage = 12;
                    let current = 0.6;
                    let luxValue = light.lux !== undefined ? light.lux : 100;

                    let sensorFailed = false;
                    if (i === 4) {
                        sensorFailed = true;
                        luxValue = -1;
                    } else {
                        sensorFailed = checkSensorFailure(i, luxValue);
                    }

                    if (sensorFailed || luxValue === -1) {
                        document.getElementById(lux-${i}).textContent = 'N/A';
                        luxValue = 0;
                        
                        if (i === 4) {
                            document.getElementById(sensor-alert-${i}).classList.add('show');
                            document.getElementById(light-card-${i}).classList.add('sensor-failed');
                        }
                    } else {
                        document.getElementById(lux-${i}).textContent = luxValue;
                        totalLux += luxValue;
                    }

                    document.getElementById(voltage-${i}).textContent = ${voltage}V;
                    document.getElementById(current-${i}).textContent = ${current}A;

                    totalVoltage += voltage;
                    totalCurrent += current;

                    const statusBadge = document.getElementById(status-${i});
                    statusBadge.className = isOn ? 'status-badge status-on' : 'status-badge status-off';
                    statusBadge.textContent = light.status;
                }
            }

            document.getElementById('total-voltage').innerHTML = ${totalVoltage}<span class="stat-unit">V</span>;
            document.getElementById('total-current').innerHTML = ${totalCurrent.toFixed(1)}<span class="stat-unit">A</span>;
            document.getElementById('total-lux').innerHTML = ${totalLux}<span class="stat-unit">Lux</span>;
            document.getElementById('system-status').textContent = 'No Fault';
            document.getElementById('system-status').style.color = '#00ff88';
        }

        function updateCharts() {
            if (!isBackendConnected) return;

            const now = new Date();
            const timeStr = now.toLocaleTimeString();

            if (timeLabels.length >= 10) {
                timeLabels.shift();
                voltageHistory.shift();
                currentHistory.shift();
            }

            timeLabels.push(timeStr);
            voltageHistory.push(parseFloat(generateFluctuatingVoltage()));
            currentHistory.push(parseFloat(generateFluctuatingCurrent()));

            if (voltageChart) {
                voltageChart.data.labels = timeLabels;
                voltageChart.data.datasets[0].data = voltageHistory;
                voltageChart.update('none');
            }

            if (currentChart) {
                currentChart.data.labels = timeLabels;
                currentChart.data.datasets[0].data = currentHistory;
                currentChart.update('none');
            }
        }

        document.addEventListener('DOMContentLoaded', function() {
